from google.cloud import bigquery
client = bigquery.Client()
sql = """
SELECT
    mother_age,
    COUNT(1) AS count,
    year
FROM
    `bigquery-public-data.samples.natality`
WHERE
    year>1997 AND year < 2009
GROUP BY
    mother_age, year
ORDER BY
    count DESC
"""
df = client.query(sql).to_dataframe()
